// ALARM TESTING - Update your testAlarm() function
// ============================================

function testAlarm(alarmType) {
    let melody = '';
    
    switch(alarmType) {
        case 'high':
            melody = document.getElementById('alarm_high_melody').value || 
                     'High:d=4,o=5,b=100:8c,8e,8g,8c6';
            break;
        case 'low':
            melody = document.getElementById('alarm_low_melody').value || 
                     'Low:d=4,o=5,b=100:8a,8c6,8e6';
            break;
        case 'urgent_low':
            melody = document.getElementById('alarm_urgent_low_melody').value || 
                     'Urgent:d=4,o=5,b=150:8c,8c,8c,8c';
            break;
        case 'no_data':
            melody = document.getElementById('alarm_no_data_melody').value || 
                     'no_data:d=4,o=5,b=100:8c,8e,8g,8c6,4p,8c,8e,8g,8c6';
            break;
    }
    
    // Send test request to device
    fetch('/api/alarm/custom', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ rtttl: melody })
    })
    .then(response => response.json())
    .then(data => {
        if (data.status === 'ok') {
            showNotification('Playing test sound...', 'success');
        } else {
            showNotification('Failed to play test sound', 'error');
        }
    })
    .catch(error => {
        showNotification('Error: ' + error.message, 'error');
    });
}
</script>
